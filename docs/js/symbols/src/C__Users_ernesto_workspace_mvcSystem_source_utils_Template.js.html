  9 		var result = ''
 10 			, source = str
 11 			, match
 12 		;
 13 		replacement = _prepareReplacement(replacement);
 14 
 15 		while (source.length > 0) {
 16 			if (match = source.match(pattern)) {
 17 				result += source.slice(0, match.index);
 18 				result += (replacement(match)) || '';
 19 				source  = source.slice(match.index + match[0].length);
 20 			} else {
 21 				result += source, source = '';
 22 			}
 23 		}
 24 		return result;
 25 	}

 27 	function _prepareReplacement(replacement) {
 28 		if (typeof(replacement)==="function") return replacement;
 29 		var template = new _Template(replacement);
 30 		return function(match) { return template.evaluate(match) };
 31 	}

 55 		_this.evaluate = function(object) {
 56 			if (typeof(object.get)==="function")
 57 				object = object.get();
 58 
 59 			return _gsub(_template, _pattern, function(match) {
 60 				if (object == null) return '';
 61 
 62 				var before = match[1] || '';
 63 				if (before == '\\') return match[2];
 64 
 65 				var ctx = object, expr = match[3];
 66 				var pattern = /^([^.[]+|\[((?:.*?[^\\])?)\])(\.|\[|$)/;
 67 				match = pattern.exec(expr);
 68 				if (match == null) return before;
 69 
 70 				while (match != null) {
 71 					var comp = (match[1].indexOf('[')===0) ? _gsub(match[2], '\\\\]', ']') : match[1];
 72 					ctx = ctx[comp];
 73 					if (null == ctx || '' == match[3]) break;
 74 					expr = expr.substring('[' == match[3] ? match[1].length : match[0].length);
 75 					match = pattern.exec(expr);
 76 				}
 77 				var _ctx = ctx || '';
 78 				return before + _ctx;
 79 			});
 80 		}
